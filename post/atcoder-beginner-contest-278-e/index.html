AtCoder Beginner Contest 278 D~E

D - All Assign Point Add

思路：区间修改+区间覆盖+单点查询，直接上线段树即可。

#include <bits/stdc++.h>
using namespace std;
template <class T> constexpr T inf = std::numeric_limits<T>::max() / 2;
using ll = long long;

const int N = 2e5 + 10;

struct Info {
	ll val;
	Info(const ll v = 0) : val(v) {}
	friend Info operator + (const Info &a, const Info &b) {
		return a.val + b.val;
	}
};
struct Node {
	Info val;
	ll lz_a, lz_d;
	Node() : val(), lz_a(0), lz_d(-1) {}
} seg[N << 2];
int a[N];

void rise(int id) {
	seg[id].val = seg[id * 2].val + seg[id * 2 + 1].val;
}
void update1(int l, int r, int id, ll d) {
	seg[id].val.val += (r - l + 1) * d;
	seg[id].lz_a += d;
}
void update2(int l, int r, int id, int d) {
	seg[id].val.val = (r - l + 1) * d;
	seg[id].lz_d = d;
	seg[id].lz_a = 0;
}
void down(int l, int r, int id) {
	int mid = l + ((r - l) >> 1);
	if (seg[id].lz_d != -1) {
		auto &d = seg[id].lz_d;
		update2(l, mid, id * 2, d);
		update2(mid + 1, r, id * 2 + 1, d);
		d = -1;
	}
	if (seg[id].lz_a != 0) {
		auto &d = seg[id].lz_a;
		update1(l, mid, id * 2, d);
		update1(mid + 1, r, id * 2 + 1, d);
		d = 0;
	}
}

void build(int l, int r, int id) {
	if (l == r) {
		seg[id].val = a[l];
	} else {
		int mid = l + ((r - l) >> 1);
		build(l, mid, id * 2);
		build(mid + 1, r, id * 2 + 1);
		rise(id);
	}
}
void add(int l, int r, int id, int al, int ar, ll d) {
	if (l == al && r == ar) {
		// seg[id].val.val += (r - l) * d;
		update1(l, r, id, d);
	} else {
		int mid = l + ((r - l) >> 1);
		down(l, r, id);
		if (ar <= mid) {
			add(l, mid, id * 2, al, ar, d);
		} else if (al > mid) {
			add(mid + 1, r, id * 2 + 1, al, ar, d);
		} else {
			add(l, mid, id * 2, al, mid, d);
			add(mid + 1, r, id * 2 + 1, mid + 1, ar, d);
		}
		rise(id);
	}
}
void modify(int l, int r, int id, int ml, int mr, ll d) {
	if (l == ml && r == mr) {
		update2(l, r, id, d);
	} else {
		int mid = l + ((r - l) >> 1);
		down(l, r, id);
		if (mr <= mid) {
			modify(l, mid, id * 2, ml, mr, d);
		} else if (ml > mid) {
			modify(mid + 1, r, id * 2 + 1, ml, mr, d);
		} else {
			modify(l, mid, id * 2, ml, mid, d); modify(mid + 1, r, id * 2 + 1, mid + 1, mr, d);
		}
		rise(id);
	}
}
Info query(int l, int r, int id, int ql, int qr) {
	if (l == ql && r == qr) {
		return seg[id].val;
	} else {
		int mid = l + ((r - l) >> 1);
		down(l, r, id);
		if (qr <= mid) {
			return query(l, mid, id * 2, ql, qr);
		} else if (ql > mid) {
			return query(mid + 1, r, id * 2 + 1, ql, qr);
		} else {
			return query(l, mid, id * 2, ql, mid) + query(mid + 1, r, id * 2 + 1, mid + 1, qr);
		}
	}
}

void Solve() {
	int n;
	cin >> n;
	for (int i = 1; i <= n; i ++) {
		cin >> a[i];
	}
	build(1, n, 1);
	int q;
	cin >> q;
	while (q --) {
		int opt;
		cin >> opt;
		if (opt == 1) {
			ll  x;
			cin >> x;
			modify(1, n, 1, 1, n, x);
		} else if (opt == 2) {
			int i;
			ll x;
			cin >> i >> x;
			add(1, n, 1, i, i, x);
		} else {
			int i;
			cin >> i;
			cout << query(1, n, 1, i, i).val << '\n';
		}
	}
}

int32_t main() {
	std::cin.tie(nullptr)->sync_with_stdio(false);
	std::cout << std::fixed << std::setprecision(12);
	int t = 1;
	// std::cin >> t;
	for (int rp = 1; t --> 0; rp ++) {
		Solve();
	}
}

E - Grid Filling

思路：我们要做的，就是找个方法能够快速去统计那个阴影矩形内的信息，越快越好。这时候，自然可以想到二维前缀和。所以，我们只需要做一次二维前缀和的预处理，然后直接计算每个数字出现的次数就行了。

void Solve() {
	int N, M, I, n, m;	
	cin >> N >> M >> I >> n >> m;
	vector a(N + 1, vector<int>(M + 1));
	vector cnt(N + 1, vector(M + 1, vector(I + 1, 0)));
	for (int i = 1; i <= N; i ++) {
		for (int j = 1; j <= M; j ++) {
			cin >> a[i][j];
			cnt[i][j][a[i][j]] ++;
		}
	}
	for (int i = 1; i <= N; i ++) {
		for (int j = 1; j <= M; j ++) {
			for (int k = 1; k <= I; k ++) {
				cnt[i][j][k] += cnt[i - 1][j][k] + cnt[i][j - 1][k] - cnt[i - 1][j - 1][k];
			}
			debug(i, j, cnt[i][j]);
		}
	}

	for (int i = n; i <= N; i ++) {
		for (int j = m; j <= M; j ++) {
			int ans = {};
			for (int k = 1; k <= I; k ++) {
				int d = cnt[i][j][k] - cnt[i - n][j][k] - cnt[i][j - m][k] + cnt[i - n][j - m][k];
				// cerr << d << ' ' << '|' << " ";
				d = cnt[N][M][k] - d;
				// cerr << d << ' ';
				if (d > 0) {
					ans ++;
				}
			}
			// cerr << '\n';
			cout << ans << " ";
		}
		cout << '\n';
	}
}
