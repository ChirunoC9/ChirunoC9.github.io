重链剖分

2022-11-15 Data Structures

重链剖分

简介

树链剖分，用于将树分割为若干条链的形式，以维护树上的信息。换句话说，将整棵树以一种方法剖分为若干条链，从而使得其可以重新组成线性结构，然后利用其他数据结构维护信息。

一般不特别说明的话，树链剖分，一般指的是重链剖分。

这里只介绍重链剖分。

重链剖分

这里先明确一些概念：

  • 重儿子：子节点中子树节点个数最大的节点。如果有多个的话，取其中一个。如果没有子节点，那么就没有重儿子。
  • 轻儿子：子节点中除了重儿子以外的其他节点。
  • 重边：父亲节点与重儿子相连的边。
  • 轻边：父亲节点和轻儿子相连的边。
  • 重链：由多条重边连接的路径。注意：落单的节点一般算作单独一条重链。
  • 轻链：由多条轻边连接的路径。

前置数据

下面是重链剖分需要用到的一些数据。

int dad[N];  // 父亲节点
int dep[N];  // 深度
int siz[N];  // 子树节点个数
int son[N];  // 重儿子
int top[N];  // 所在重链的顶部节点
int dfn[N];  // dfs序列
int rnk[N];  // 记录dfs序列对应的节点编号

如何实现

树及树剖的代码实现。

vector<int> edge[N];
int dad[N], dep[N], siz[N], son[N], top[N] dfn[N], rnk[N], stamp = 0;

重链剖分需要两次dfs，以实现预处理。

第一遍dfs需要实现下面的操作：

  • 找到节点所在子树的大小
  • 找到其重儿子
  • 找到其父亲
  • 处理其深度
void dfs1(int from) {
    son[from] = -1;
    siz[from] = 1;
    for (auto to : edge[from]) {
        if (dep[to] == -1) {  // 实际上只要子节点没有被访问过就行
            dep[to] == dep[from] + 1;
            dad[to] = from;
            dfs1(to);
            siz[from] += siz[to];
            if (son[from] == -1 || siz[to] > siz[son[from]]) {
                son[from] = to;
            }
        }
    }
}

第二遍dfs需要实现下面的操作：

  • 连接重链
  • 记录重链的链顶
  • 记录每个节点的dfs序
  • 记录dfs序对应的节点编号
void dfs2(int from, int root) {
    top[from] = root;
    dfn[from] = ++ stamp;
    rnk[stamp] = from;
    if (son[from] == -1) {
        return;
    }
    dfs2(son[from], root);
    for (auto to : edge[from]) {
        if (to != son[from] && to != dad[from]) {
            dfs2(to, to);
        }
    }
}

之后，就可以根据dfs序，以合适的数据结构进行维护了。

// 树剖启动
void build(int root) {
    memset(dep, -1, sizeof dep);
    dep[root] = 0;
    dfs1(root);
    dfs2(root, root);
}

重链剖分的性质

  1. 重链剖分是将整棵树完全剖分。
  2. 树上每个点都属于且仅属于一条重链。
  3. 剖分时重链优先遍历，且重链内的dfs序是有序的。
  4. 向下经过一条轻边时，所在子树的大小至少会除以222。
  5. 从任意节点走到根节点，经过轻边的数量最多不会超过log⁡n\log{n}logn条。

利用第三条性质，再加上LCA，就可以利用一些维护区间的数据结构，对树进行维护。

LCA

树上两个点，如果在同一条重链，深度深的就是公共祖先。

如果不在的话，那么深度深的先跳，直到跳到同一条链上。

根据上面第五个性质，可以发现其实每次就是跳出一条重链，并经过一条轻链。因此，其时间复杂度为O(log⁡n)O(\log{n})O(logn)。

int lca(int a, int b) {
    while (top[a] != top[b]) {
        if (dep[top[a]] < dep[top[b]]) {
            swap(a, b);
        } 
        a = fa[top[a]];
    }
    if (dep[a] < dep[b]) {
        return a;
    } else {
        return b;
    }
}

维护树链

同一条重链的dfs序是连续的，因此可以使用一些数据结构（如线段树）进行维护。下面是两道板子题，通过这两道题可以看出线段树是如何维护树链的性质的。

P2146 [NOI2015] 软件包管理器

题目传送门

知识点：线段树维护点权。

对于install操作，也就是所有这个节点的祖先都需要被安装，因此，可以从这个节点的重链上起跳，直到跳到根节点链为止。显然，如果直接对链维护，时间复杂度为O(log⁡n)O(\log{n})O(logn)，如果其中是使用线段树进行维护的话，那么还有O(log⁡n)O(\log{n})O(logn)的时间复杂度（显然比暴力维护好）。

对于uninstall操作，那么很显然，所有依赖于这份节点的儿子节点就会被卸掉。什么软件会被卸载掉呢？很明显，就是这个节点的的子树，所有的子树节点都会被标记为未安装状态。

依据上面的思路的话，那么只需要对原先的树进行一次树剖就可以了。对应的线段树，需要的是一种类似于取反操作的线段树。同时，因为每次操作都需要一个返回软件包数量，因此，在修改的过程中，也可以附加上计数这个操作。

/* P2146 [NOI2015] 软件包管理器 */
#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 10;

vector<int> edge[N];
int dep[N], son[N], fa[N], top[N], siz[N], dfn[N], rnk[N];
int stamp = 0;

int n;

void add_edge(int from, int to) {
    edge[from].push_back(to);
    edge[to].push_back(from);
}

void build(int root) {
    function<void(int)> dfs1 = [&](int from) {
        son[from] = -1;
        siz[from] = 1;
        for (auto to : edge[from]) {
            if (!dep[to]) {
                dep[to] = dep[from] + 1;
                fa[to] = from;
                dfs1(to);
                siz[from] += siz[to];
                if (son[from] == -1 || siz[to] > siz[son[from]]) {
                    son[from] = to;
                }
            }
        }
    };
    function<void(int,int)> dfs2 = [&](int from, int root) {
        dfn[from] = ++ stamp;
        rnk[from] = stamp;
        top[from] = root;
        if (son[from] == -1) {
            return;
        }
        dfs2(son[from], root);
        for (auto to : edge[from]) {
            if (to != son[from] && to != fa[from]) {
                dfs2(to, to);
            }
        }
    };

    dep[root] = 1; 
    dfs1(root);
    dfs2(root, root);
}

int seg[N << 2], lazy[N << 2];
void build(int l, int r, int id) {
    if (l == r) {
        lazy[id] = -1;
    } else {
        int mid = l + ((r - l) >> 1);
        build(l, mid, id * 2);
        build(mid + 1, r, id * 2 + 1);
        lazy[id] = -1;
    }
}
void rise(int id) {
    seg[id] = seg[id * 2] + seg[id * 2 + 1];
}
void down(int l, int r, int id) {
    auto &d = lazy[id];
    if (d == -1) {
        return;
    }
    int mid = l + ((r - l) >> 1);
    seg[id * 2] = (mid - l + 1) * d;
    seg[id * 2 + 1] = (r - mid) * d;
    lazy[id * 2] = d;
    lazy[id * 2 + 1] = d;
    d = -1;
}
int change(int l, int r, int id, int sl, int sr, int d) {
    if (l == sl && r == sr) {
        int ret = seg[id];
        seg[id] = (r - l + 1) * d;
        lazy[id] = d;
        ret = abs(ret - seg[id]);
        return ret;
    } else {
        int mid = l + ((r - l) >> 1);
        int ret = 0;
        down(l, r, id);
        if (sr <= mid) {
            ret = change(l, mid, id * 2, sl, sr, d);
        } else if (sl > mid) {
            ret = change(mid + 1, r, id * 2 + 1, sl, sr, d);
        } else {
            ret = change(l, mid, id * 2, sl, mid, d) + change(mid + 1, r, id * 2 + 1, mid + 1, sr, d);
        }
        rise(id);
        return ret;
    }
}

int opt1(int x) {
    int ret = 0;
    while (dep[top[x]] > 1) {
        ret += change(1, n, 1, dfn[top[x]], dfn[x], 1);
        x = fa[top[x]];
    }
    ret += change(1, n, 1, dfn[top[x]], dfn[x], 1);
    return ret;
}

int opt2(int x) {
    return change(1, n, 1, dfn[x], dfn[x] + siz[x] - 1, 0);
}

void Solution() {
    cin >> n;
    for (int i = 1; i < n; ++ i) {
        int x;
        cin >> x;
        add_edge(x, i);
    }
    build(0);

    int q;
    cin >> q;
    while (q --) {
        string opt;
        cin >> opt;
        if (opt == "install") {
            int x;
            cin >> x;
            cout << opt1(x) << '\n';
        } else {
            int x;
            cin >> x;
            cout << opt2(x) << '\n';
        }
    }
}

int main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    Solution();
}

P2420 让我们异或吧

题目传送门

知识点：线段树维护边权。

题意简述一下，就是给定一棵树，需要对每次询问，求出两点之间路径上所有边权的异或和。

树链剖分最擅长的事情就是解决树上路径的事情了。但是，本题中要解决的却是边权的异或和，而树链剖分解决的，却是点权问题。

这时候就需要将边权下放，如父亲节点连向儿子节点的边，其边权给儿子节点。这样，就完成了边权转化为点权的任务了。在之后的查询过程中，由于lca节点代表的边权其实并不属于要查询的路径，那么对于dfs序列来说，不查lca相当于lca节点的dfn序+1，也就是最多查到lca节点的子儿子（如果两个节点跳到同一个节点的话，那么这个节点就不查了）。这样，前置树链剖分就完成了。

接下来是根据dfn序的点值建维护区间异或和的线段树，查询就解决了。

/* P2420 让我们异或吧 */
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 10;
vector<pair<int,int>> edge[N];

int top[N], siz[N], fa[N], dep[N], son[N], dfn[N];
int stamp = 0;
int w[N], _w[N];

void add_edge(int from, int to, int w) {
    edge[from].push_back({to, w});
    edge[to].push_back({from, w});
}

void dfs1(int from, int w) {
    siz[from] = 1;
    son[from] = -1;
    _w[from] = w;
    for (auto [to, _w] : edge[from]) {
        if (!dep[to]) {
            dep[to] = dep[from] + 1;
            fa[to] = from;
            dfs1(to, _w);
            siz[from] += siz[to];
            if (son[from] == -1 || siz[to] > siz[son[from]]) {
                son[from] = to;
            }
        }
    }
}
void dfs2(int from, int root) {
    dfn[from] = ++ stamp;
    top[from] = root;
    w[dfn[from]] = _w[from];
    if (son[from] == -1) {
        return;
    }
    dfs2(son[from], root);
    for (auto [to, _] : edge[from]) {
        if (to != son[from] && to != fa[from]) {
            dfs2(to, to);
        }
    }
}
void build(int root) {
    dep[root] = 1;
    dfs1(root, 0);
    dfs2(root, root);
}

int n;
int seg[N << 2];

void build(int l, int r, int id) {
    if (l == r) {
        seg[id] = w[l];
    } else {
        int mid = l + ((r - l) >> 1);
        build(l, mid, id * 2);
        build(mid + 1, r, id * 2 + 1);
        seg[id] = seg[id * 2] ^ seg[id * 2 + 1];
    }
}
int query(int l, int r, int id, int ql, int qr) {
    if (l == ql && r == qr) {
        return seg[id];
    } else {
        int mid = l + ((r - l) >> 1);
        if (qr <= mid) {
            return query(l, mid, id * 2, ql, qr);
        } else if (ql > mid) {
            return query(mid + 1, r, id * 2 + 1, ql, qr);
        } else {
            return query(l, mid, id * 2, ql, mid) ^ query(mid + 1, r, id * 2 + 1, mid + 1, qr);
        }
    }
}
int query(int a, int b) {
    int ret = 0;
    while (top[a] != top[b]) {
        if (dep[top[a]] < dep[top[b]]) {
            swap(a, b);
        }
        ret ^= query(1, n, 1, dfn[top[a]], dfn[a]);
        a = fa[top[a]];
    }
    if (dep[a] > dep[b]) {
        swap(a, b);
    }
    if (dep[a] < dep[b]) {
        ret ^= query(1, n, 1, dfn[a] + 1, dfn[b]);
    }
    return ret;
}

void Solution() {
    cin >> n;
    for (int i = 1; i < n; ++ i) {
        int a, b, w;
        cin >> a >> b >> w;
        add_edge(a, b, w);
    }
    build(1);
    build(1, n, 1);

    int q;
    cin >> q;
    while (q --) {
        int a, b;
        cin >> a >> b;
        cout << query(a, b) << '\n';
    }
}

int32_t main() {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    Solution();
}
